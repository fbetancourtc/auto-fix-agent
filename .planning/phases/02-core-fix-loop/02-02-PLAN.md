**2. Create `scripts/validate-diff.sh`:**

Follow the same pattern as `scripts/sanitize-logs.sh` (existing convention).

```bash
#!/usr/bin/env bash
set -euo pipefail

# validate-diff.sh — Post-agent diff validation for FIXG-02 / FIXG-03
# Checks git diff against allowed_dirs from repo-stack-map.json.
# Reverts any forbidden file changes. Exits with status indicating result.
#
# Usage: validate-diff.sh <repo> <config-path>
# Exit codes: 0 = clean (all changes allowed or forbidden files reverted)
#             1 = no changes detected (agent produced no diff)

REPO="${1:?Usage: validate-diff.sh <repo> <config-path>}"
CONFIG="${2:?Usage: validate-diff.sh <repo> <config-path>}"

# Get allowed directories for this repo
# First check for per-repo override, then fall back to stack default
STACK=$(jq -r --arg repo "$REPO" '.repos[$repo].stack // empty' "$CONFIG")
if [ -z "$STACK" ]; then
  echo "ERROR: Repo '$REPO' not found in config"
  exit 1
fi

REPO_DIRS=$(jq -r --arg repo "$REPO" '.repos[$repo].allowed_dirs // empty | .[]?' "$CONFIG" 2>/dev/null)
if [ -n "$REPO_DIRS" ]; then
  ALLOWED_DIRS="$REPO_DIRS"
else
  ALLOWED_DIRS=$(jq -r --arg stack "$STACK" '.defaults[$stack].allowed_dirs[]' "$CONFIG")
fi

if [ -z "$ALLOWED_DIRS" ]; then
  echo "ERROR: No allowed_dirs found for repo '$REPO' (stack: $STACK)"
  exit 1
fi

# Get list of changed files (staged and unstaged)
CHANGED_FILES=$(git diff --name-only HEAD 2>/dev/null || true)
STAGED_FILES=$(git diff --cached --name-only 2>/dev/null || true)
ALL_CHANGES=$(echo -e "${CHANGED_FILES}\n${STAGED_FILES}" | sort -u | grep -v '^$' || true)

if [ -z "$ALL_CHANGES" ]; then
  echo "WARNING: No file changes detected -- agent produced no diff"
  exit 1
fi

VIOLATIONS=()
ALLOWED_COUNT=0

while IFS= read -r file; do
  MATCH=false
  while IFS= read -r dir; do
    if [[ "$file" == "$dir"* ]]; then
      MATCH=true
      break
    fi
  done <<< "$ALLOWED_DIRS"

  if [ "$MATCH" = false ]; then
    VIOLATIONS+=("$file")
    git checkout HEAD -- "$file" 2>/dev/null || true
  else
    ALLOWED_COUNT=$((ALLOWED_COUNT + 1))
  fi
done <<< "$ALL_CHANGES"

echo "Diff validation: ${ALLOWED_COUNT} allowed file(s)"

if [ ${#VIOLATIONS[@]} -gt 0 ]; then
  echo "::warning::Reverted ${#VIOLATIONS[@]} forbidden file(s): ${VIOLATIONS[*]}"
  for v in "${VIOLATIONS[@]}"; do
    echo "  REVERTED: $v"
  done
fi

if [ "$ALLOWED_COUNT" -eq 0 ]; then
  echo "WARNING: All changes were in forbidden directories -- nothing to commit"
  exit 1
fi

echo "VALID: ${ALLOWED_COUNT} file(s) in allowed directories"
exit 0
```

Make the script executable: `chmod +x scripts/validate-diff.sh`

Key design decisions (Claude's discretion from CONTEXT.md):
- Central script (follows `sanitize-logs.sh` pattern)
- Per-repo `allowed_dirs` override falls back to stack defaults
- Exit code 1 means "no valid changes" (agent fix was all in forbidden dirs or produced no diff)
- Forbidden files are reverted with `git checkout HEAD --`
- The `::warning::` syntax creates a GitHub Actions annotation

**5. Add push and PR creation step (new step 10):**

After diff validation passes, the workflow pushes the agent's commit and creates the PR:

```yaml
# 10. Push fix branch and create PR (PRMG-01, PRMG-02)
# Workflow controls push + PR creation after diff validation passes.
- name: Push and create PR
  id: create-pr
  if: steps.circuit.outputs.should_proceed == 'true' && steps.flaky.outputs.is_flaky != 'true' && steps.diff-check.outcome == 'success'
  env:
    GH_TOKEN: ${{ steps.app-token.outputs.token }}
    RUN_ID: ${{ inputs.failed_run_id }}
    TARGET_REPO: ${{ inputs.repository }}
    HEAD_BRANCH: ${{ inputs.head_branch }}
  run: |
    BRANCH_NAME="auto-fix/${RUN_ID}"

    # Check if agent created a branch and has commits
    CURRENT_BRANCH=$(git branch --show-current)
    if [ "$CURRENT_BRANCH" != "$BRANCH_NAME" ]; then
      # Agent may not have created the branch -- create it from current state
      git checkout -b "$BRANCH_NAME" 2>/dev/null || true
    fi

    # Stage any remaining changes (validate-diff.sh may have reverted some)
    git add -A
    CHANGES=$(git diff --cached --name-only)
    if [ -z "$CHANGES" ]; then
      echo "No changes to push after diff validation"
      exit 0
    fi

    # Amend the agent's commit to include only validated changes (if reverts happened)
    git diff --cached --quiet || git commit --amend --no-edit 2>/dev/null || git commit -m "fix: auto-fix for CI failure (run ${RUN_ID})"

    # Push the fix branch
    git push origin "$BRANCH_NAME" --force-with-lease

    # Create PR targeting develop
    PR_URL=$(gh pr create \
      --repo "$TARGET_REPO" \
      --base develop \
      --head "$BRANCH_NAME" \
      --label "auto-fix" \
      --title "fix: auto-fix for CI failure (run #${RUN_ID})" \
      --body "## Auto-Fix PR

**Triggered by:** CI failure in run [#${RUN_ID}](https://github.com/${TARGET_REPO}/actions/runs/${RUN_ID})
**Branch:** \`${HEAD_BRANCH}\`

This PR was automatically generated by the auto-fix agent.

### Review Checklist
- [ ] Changes are limited to source code only
- [ ] Fix addresses the root cause identified in the CI logs
- [ ] Tests pass with the applied fix

---
*Generated by [auto-fix-agent](https://github.com/fbetancourtc/auto-fix-agent)*")

    echo "pr_url=$PR_URL" >> "$GITHUB_OUTPUT"
    echo "Created PR: $PR_URL"
```

NOTE on PR description: The agent will enhance this template through prompt instructions (Plan 02-03). This step provides a baseline PR body. The agent's detailed root cause analysis and verification results are captured in its commit message (via prompt instructions).

NOTE on --force-with-lease: Used instead of --force for safety. Since this is a new branch, it's effectively the same as a normal push, but protects against race conditions.
